  2. ; tilegen by jimi (malcolmj1@juno.com)

  3. ;  draws 16x8 tile map

  4. ;======================================

  5. tilegen:

  6. 	ld ix,$fc00

  7. 	ld b,8

  8. loop_row:

  9. 		push bc

  10. columns:

  11. 	ld b,16

  12. loop_columns:

  13. 		push hl

  14. 	ld l,(hl)

  15. 	ld h,0

  16. 	add hl,hl

  17. 	add hl,hl

  18. 	add hl,hl

  19. 

  20. 	ld de,tile0

  21. 	add hl,de

  22. 

  23. draw_tile:

  24. 	ld de,$10

  25. 	ld c,b

  26. 	ld b,8

  27. 

  28. 		push ix

  29. draw_tile_loop:

  30. 	ld a,(hl)

  31. 	ld (ix),a

  32. 	add ix,de

  33. 

  34. 	inc hl

  35. 

  36. 	djnz draw_tile_loop

  37. 		pop ix

  38. 

  39. 	ld b,c

  40. 		pop hl

  41. 

  42. 	inc hl

  43. 	inc ix

  44. 	djnz loop_columns

  45. 	ld de,$10*7

  46. 	add ix,de

  47. 		pop bc

  48. 	djnz loop_row

  49. 	ret
